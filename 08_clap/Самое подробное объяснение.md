// ┌─────────────────────────────────────────────────────────────────────────┐
// │                          ИМПОРТ НЕОБХОДИМЫХ ТРЕЙТОВ                      │
// └─────────────────────────────────────────────────────────────────────────┘

// clap::Parser — это трейт (набор методов), который добавляет структуре
// возможность парсить аргументы командной строки. Без него нельзя вызвать
// .parse() на нашей структуре Cli.
use clap::Parser;

// clap::ValueEnum — трейт для перечислений (enum). Он говорит clap:
// "значение этого enum должно быть одним из перечисленных вариантов".
// Благодаря этому clap сам проверит, что пользователь ввёл допустимое значение,
// и выдаст ошибку с подсказкой (например, "possible values: ru, en").
use clap::ValueEnum;

// ┌─────────────────────────────────────────────────────────────────────────┐
// │                         ПЕРЕЧИСЛЕНИЕ Language                           │
// └─────────────────────────────────────────────────────────────────────────┘

// #[derive(ValueEnum, Clone, Debug)] — это атрибуты, которые заставляют компилятор
// автоматически реализовать для enum Language указанные трейты.
// - ValueEnum: как сказано выше, позволяет clap проверять допустимые варианты.
// - Clone: позволяет копировать значение enum (нужно для default_value_t, потому что
//   clap создаёт копию значения по умолчанию).
// - Debug: позволяет печатать значение через {:?}, полезно для вывода в verbose-режиме.
#[derive(ValueEnum, Clone, Debug)]
enum Language {
    Ru,   // вариант "ru" (команда --language ru)
    En,   // вариант "en"
    // De, // можно добавить, но пока закомментировано
    // It, // можно добавить
}

// ┌─────────────────────────────────────────────────────────────────────────┐
// │                    СТРУКТУРА Cli — ОПИСАНИЕ АРГУМЕНТОВ                  │
// └─────────────────────────────────────────────────────────────────────────┘

// #[derive(Parser)] — это главная "магия". Компилятор генерирует код,
// который умеет читать аргументы командной строки и заполнять поля структуры.
// Аналогия: вы заполняете бланк заказа (структура), а clap берёт на себя роль
// оператора, который проверяет, что вы всё ввели правильно.
#[derive(Parser)]
// #[command(...)] — даёт метаинформацию о программе. Эти данные попадают в --help.
#[command(
    name = "greet",           // имя программы (показывается в строке Usage)
    version = "1.0",          // версия (выводится по флагу --version)
    about = "Приветствие с валидацией" // краткое описание
)]
struct Cli {
    /// Имя (позиционный аргумент)
    // Поле name типа String. Позиционный аргумент означает, что пользователь
    // просто пишет значение после имени программы, без флага.
    // Например: ./greet Alice
    // Это обязательный аргумент (если его нет, clap выдаст ошибку).
    name: String,

    /// Язык (флаг -l или --language)
    // Атрибут #[arg(...)] настраивает парсинг для этого поля.
    // - short, long: разрешены флаги -l и --language.
    // - value_enum: указывает, что значение должно быть из enum Language.
    // - default_value_t = Language::Ru: если пользователь не указал язык,
    //   подставляется значение Language::Ru.
    //   Благодаря этому поле становится необязательным.
    #[arg(short, long, value_enum, default_value_t = Language::Ru)]
    language: Language,

    /// Город (обязательная опция -c или --city)
    // Здесь нет default_value_t, и тип не Option => clap считает этот аргумент
    // обязательным. Пользователь обязан написать --city Москва или -c Москва.
    // Если забудет — clap сам выведет ошибку.
    #[arg(short, long)]
    city: String,

    /// Возраст (опция -a или --age, только 0..120)
    // Тип Option<u8> означает, что этот аргумент необязательный.
    // value_parser = clap::value_parser!(u8).range(0..=120) — это валидатор:
    //   - сначала значение парсится как целое число u8,
    //   - затем проверяется, что оно входит в диапазон 0..120.
    // Если пользователь введёт число вне диапазона, clap выдаст ошибку.
    #[arg(short, long, value_parser = clap::value_parser!(u8).range(0..=120))]
    age: Option<u8>,

    /// Подробный вывод (флаг -v или --verbose)
    // Поле типа bool. Если флаг -v или --verbose присутствует, verbose = true.
    // Если флаг отсутствует, verbose = false (по умолчанию). Это стандартный булев флаг.
    // Аналогия: переключатель, который включает "режим отладки".
    #[arg(short, long)]
    verbose: bool,
}

// ┌─────────────────────────────────────────────────────────────────────────┐
// │                            ФУНКЦИЯ main                                │
// └─────────────────────────────────────────────────────────────────────────┘

fn main() {
    // let args = Cli::parse(); — ВОЛШЕБНАЯ СТРОКА.
    // Cli::parse() делает следующее:
    // 1) Читает аргументы из std::env::args() (то, что пользователь ввёл в терминале).
    // 2) Согласно описанию структуры (поля и атрибуты) разбирает эти аргументы.
    // 3) Если произошла ошибка (не хватает обязательного аргумента,
    //    неверный тип, значение вне диапазона, неизвестный флаг), то clap:
    //    - печатает понятное сообщение об ошибке
    //    - печатает справочную информацию (--help)
    //    - завершает программу с кодом ошибки (программа до этой строки не доходит).
    // 4) Если всё хорошо, возвращает экземпляр Cli с заполненными полями.
    let args = Cli::parse();

    // Выбираем строку приветствия в зависимости от языка.
    // args.language — это значение enum Language (Ru, En и т.д.).
    // match сравнивает его с вариантами и возвращает соответствующий &str.
    // Аналогия: посмотрели на какой язык переключён тумблер, выбрали фразу.
    let greeting = match args.language {
        Language::Ru => "Привет",
        Language::En => "Hello",
        // Language::De => "Hallo",
        // Language::It => "Ciao",
        // Если добавить другие языки, нужно дописать ветки.
    };

    // Проверяем, включён ли подробный режим (verbose).
    // args.verbose — это bool: true если пользователь указал -v или --verbose.
    if args.verbose {
        // Выводим отладочную информацию.
        // Здесь println! с префиксом [VERBOSE] для наглядности.
        // {:?} — формат для отладки (Debug). Так как у Language есть #[derive(Debug)],
        // можно напечатать его значение (например, "Ru" или "En").
        println!("[VERBOSE] Язык: {:?}", args.language);
        println!("[VERBOSE] Имя: {}", args.name);
        println!("[VERBOSE] Город: {}", args.city);
        
        // if let Some(age) = args.age — безопасное извлечение значения из Option.
        // Если age = Some(число), то переменная age получает это число (тип u8)
        // и выполняется блок кода. Если age = None, то блок пропускается.
        // Это более удобная форма match для случая, когда нас интересует только Some.
        if let Some(age) = args.age {
            println!("[VERBOSE] Возраст: {}", age);
        }
    }

    // Основной вывод: приветствие, имя, город.
    // greeting — выбранная фраза, args.name — имя, args.city — город.
    println!("{}, {}! Твой город: {}", greeting, args.name, args.city);
    
    // Если возраст был указан (args.age != None), выводим дополнительную фразу.
    // Используем if let Some(age) — как и выше, безопасно извлекаем значение.
    if let Some(age) = args.age {
        println!("Тебе {} лет.", age);
    }
}


// ┌─────────────────────────────────────────────────────────────────────────┐
// │            ПОЧЕМУ У ОДНИХ ПОЛЕЙ ЕСТЬ DEFAULT_VALUE, А У ДРУГИХ НЕТ?     │
// │            КАК РЕШАТЬ: ОБЯЗАТЕЛЬНО / НЕОБЯЗАТЕЛЬНО / С ДЕФОЛТОМ         │
// └─────────────────────────────────────────────────────────────────────────┘
//
// В CLI-программе каждое поле — это один аргумент. Разработчик должен решить:
//   - Должен ли пользователь всегда указывать этот аргумент? (обязательный)
//   - Или он может его пропустить, и тогда будет использовано значение по умолчанию?
//   - Или он может его пропустить, и тогда программа просто ничего не будет знать об этом?
//
// clap даёт три основных способа выразить это через типы и атрибуты.

// ┌─────────────────────────────────────────────────────────────────────────┐
// │ 1. ОБЯЗАТЕЛЬНЫЙ АРГУМЕНТ (без default, без Option)                     │
// └─────────────────────────────────────────────────────────────────────────┘
// Поле с типом `T` (не Option<T>) и без `default_value` означает:
//   - Пользователь ОБЯЗАН указать этот аргумент.
//   - Если он его не укажет, clap выдаст ошибку ещё до main и покажет справку.
//   - Пример: `name: String` (позиционный) и `city: String` (опция --city).
// Почему `city` обязательный? Потому что по замыслу программы город — важная часть информации,
// без которой приветствие не имеет смысла. Если пользователь забудет --city, это скорее всего
// ошибка, и программа не должна пытаться угадать город.

// ┌─────────────────────────────────────────────────────────────────────────┐
// │ 2. НЕОБЯЗАТЕЛЬНЫЙ АРГУМЕНТ СО ЗНАЧЕНИЕМ ПО УМОЛЧАНИЮ (default_value)   │
// └─────────────────────────────────────────────────────────────────────────┘
// Поле с `default_value` (и без `Option`) означает:
//   - Пользователь может указать аргумент, а может пропустить.
//   - Если пропускает — clap сам подставит значение по умолчанию.
//   - Пример: `language: Language` с `default_value_t = Language::Ru`.
// Почему именно так? Язык приветствия — удобно иметь значение по умолчанию (русский),
// но если пользователь захочет другой язык, он может его явно задать. Такой подход
// делает программу дружелюбной: не нужно каждый раз писать --language ru.

// ┌─────────────────────────────────────────────────────────────────────────┐
// │ 3. НЕОБЯЗАТЕЛЬНЫЙ АРГУМЕНТ БЕЗ ЗНАЧЕНИЯ ПО УМОЛЧАНИЮ (Option<T>)       │
// └─────────────────────────────────────────────────────────────────────────┘
// Поле с типом `Option<T>` (и без `default_value`) означает:
//   - Пользователь может указать аргумент, а может пропустить.
//   - Если пропускает — в программе будет `None`.
//   - Пример: `age: Option<u8>`.
// Почему не `default_value`? Возраст — это необязательная информация. Нет разумного
// значения по умолчанию (нельзя подставить 0 или 18, это будет ложью). Лучше явно
// проверить в коде: если `Some(age)` — показать фразу, если `None` — ничего не говорить.

// ┌─────────────────────────────────────────────────────────────────────────┐
// │ 4. БУЛЕВ ФЛАГ (bool) — особый случай                                   │
// └─────────────────────────────────────────────────────────────────────────┘
// Поле типа `bool` с `#[arg(short, long)]` (без default_value) работает так:
//   - Если флаг указан → true, не указан → false.
//   - Не требует значения (просто наличие флага).
//   - Пример: `verbose: bool`.
// Это логично: verbose — это переключатель, у которого есть естественное значение
// по умолчанию (false, то есть выключен).

// ┌─────────────────────────────────────────────────────────────────────────┐
// │ ТАБЛИЦА ПРИНЯТИЯ РЕШЕНИЙ (категория "что если")                        │
// └─────────────────────────────────────────────────────────────────────────┘
//
// Вопрос: "Что если пользователь не указал аргумент?"
// -------------------------------------------------------------------------
// | Тип поля                 | default_value | Результат                   |
// -------------------------------------------------------------------------
// | T                        | нет           | Ошибка, программа не запустится |
// | T                        | есть          | Будет использовано значение по умолчанию |
// | Option<T>                | нет           | args.field == None          |
// | Option<T>                | есть          | Редко нужно, но можно: тогда default_value применяется, но тип остаётся Option |
// | bool                     | -             | false, если флаг отсутствует |
//
// ┌─────────────────────────────────────────────────────────────────────────┐
// │ ПРАКТИЧЕСКИЙ СОВЕТ:                                                    │
// │ - Если аргумент критичен для логики → обязательный (без Option).       │
// │ - Если аргумент можно разумно угадать → default_value.                 │
// │ - Если аргумент нужен не всегда и нет разумного дефолта → Option<T>.   │
// │ - Если аргумент — просто переключатель → bool.                         │
// └─────────────────────────────────────────────────────────────────────────┘

// ┌─────────────────────────────────────────────────────────────────────────┐
// │ ПРИМЕРЫ ЗАПУСКА И ОШИБКИ (ЧТО ПРОИЗОЙДЁТ ПРИ РАЗНЫХ СЦЕНАРИЯХ)        │
// └─────────────────────────────────────────────────────────────────────────┘
//
// 1. Правильный запуск со всеми опциями:
//    $ cargo run -- Alice --city Moscow --language en --age 25 --verbose
//    Вывод:
//      [VERBOSE] Язык: En
//      [VERBOSE] Имя: Alice
//      [VERBOSE] Город: Moscow
//      [VERBOSE] Возраст: 25
//      Hello, Alice! Твой город: Moscow
//      Тебе 25 лет.
//
// 2. Запуск без возраста и без verbose:
//    $ cargo run -- Bob --city SPb
//    Вывод:
//      Привет, Bob! Твой город: SPb
//    (возраст не выводится, verbose молчит)
//
// 3. Запуск без обязательного --city:
//    $ cargo run -- Charlie
//    Ошибка от clap (до main):
//      error: the following required arguments were not provided: --city <CITY>
//      Usage: greet <NAME> --city <CITY> ...
//
// 4. Запуск с неверным значением языка:
//    $ cargo run -- Dave --city London --language fr
//    Ошибка от clap:
//      error: 'fr' is not a valid value for '--language <LANGUAGE>'
//      Possible values: ru, en
//
// 5. Запуск с возрастом вне диапазона:
//    $ cargo run -- Eve --city NY --age 150
//    Ошибка от clap:
//      error: value validation failed for '--age <AGE>': 150 is not in 0..=120
//
// 6. Запуск с неизвестным флагом:
//    $ cargo run -- Frank --city Berlin --verbose --unknown
//    Ошибка от clap:
//      error: unexpected argument '--unknown' found
//      ...
